# =============================================================================
# commands.txt — Quick reference for running the Agent Control Plane
# =============================================================================


# -----------------------------------------------------------------------------
# SETUP
# -----------------------------------------------------------------------------

# Install dependencies
pip install -r requirements.txt

# Copy env template and fill in your API key
cp .env.example .env


# -----------------------------------------------------------------------------
# LOCAL MODE — function tools, no MCP servers needed
# -----------------------------------------------------------------------------

# Web UI — calculator agent
adk web . --port 8000

# Web UI — palindrome agent
MANIFEST_PATH=calculator_agent/manifest/palindrome_agent_manifest.yaml adk web . --port 8000

# Terminal REPL — calculator agent
python -m calculator_agent.agent_from_manifest

# Terminal REPL — palindrome agent
MANIFEST_PATH=calculator_agent/manifest/palindrome_agent_manifest.yaml python -m calculator_agent.agent_from_manifest


# -----------------------------------------------------------------------------
# MCP MODE — tools served as standalone HTTP servers
# Start servers first, then run the agent
# -----------------------------------------------------------------------------

# Terminal 1 — calculator MCP server (port 8001)
python -m mcp_servers.calculator_server

# Terminal 2 — palindrome MCP server (port 8002)
python -m mcp_servers.palindrome_server

# Terminal 3 — agent using MCP tools (web UI)
MANIFEST_PATH=calculator_agent/manifest/calculator_mcp_manifest.yaml adk web . --port 8000

# Terminal 3 — agent using MCP tools (terminal REPL)
MANIFEST_PATH=calculator_agent/manifest/calculator_mcp_manifest.yaml python -m calculator_agent.agent_from_manifest

# Palindrome agent in MCP mode
MANIFEST_PATH=calculator_agent/manifest/palindrome_mcp_manifest.yaml python -m calculator_agent.agent_from_manifest


# -----------------------------------------------------------------------------
# DOCKER MODE
# -----------------------------------------------------------------------------

# Build image
docker compose build

# Run agent only (local function tools)
docker compose up

# Run agent + both MCP servers
# (set MANIFEST_PATH to an MCP manifest in .env first,
#  and update manifest URLs to use Docker service names:
#    calculator_mcp_manifest.yaml → url: http://calculator-mcp:8001/mcp
#    palindrome_mcp_manifest.yaml → url: http://palindrome-mcp:8002/mcp)
docker compose --profile mcp up

# Stop all containers
docker compose down

# Rebuild after code changes
docker compose build && docker compose up


# -----------------------------------------------------------------------------
# SWITCHING AGENTS
# Change MANIFEST_PATH in .env or pass it inline — no code changes needed
# -----------------------------------------------------------------------------

# Available manifests:
#   calculator_agent/manifest/calculator_agent_manifest.yaml   → calculator, local tools
#   calculator_agent/manifest/palindrome_agent_manifest.yaml   → palindrome, local tools
#   calculator_agent/manifest/calculator_mcp_manifest.yaml     → calculator, MCP tools
#   calculator_agent/manifest/palindrome_mcp_manifest.yaml     → palindrome, MCP tools
