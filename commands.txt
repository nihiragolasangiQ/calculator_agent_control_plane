# commands.txt — quick reference

# setup
pip install -r requirements.txt
cp .env.example .env
gcloud auth application-default login

# single agent (default)
adk web .
adk web . --port 8080
python -m orchestrator.agent

# orchestrator mode — auto-discovers all agents in orchestrator/manifest/
ORCHESTRATOR_MODE=true adk web .

# add a new agent: drop a *_manifest.yaml into orchestrator/manifest/ and restart
# test a single agent in isolation (bypasses orchestrator)
MANIFEST_PATH=orchestrator/manifest/my_agent_manifest.yaml adk web .

# runtime overrides — any YAML value can be set inline
AGENT_MODEL_NAME=gemini-2.5-pro adk web .
CONFIDENCE_THRESHOLD=0.9 adk web .
BQ_PROJECT=my-other-project adk web .
ENABLE_LLM_POLICY_GUARDRAIL=0 adk web .
GUARDRAIL_FAIL_CLOSED=0 adk web .

# diagnostics
adk --version
gcloud auth application-default print-access-token
bq ls gsk-corp-ai-tech-ops-dev:snow
ORCHESTRATOR_MODE=true adk web . 2>&1 | head -50
